66static inline const char *YAWT_q_cid_to_hex(const YAWT_Q_Cid_t *cid) {
67 static char buf[41];
68 memset(buf, 0, sizeof(buf));
69 for (uint8_t i = 0; i < cid->len && i < 20; i++) {
70 sprintf(buf + 2 * i, "%02x", cid->id[i]);
71 }
72 return buf;
73}
