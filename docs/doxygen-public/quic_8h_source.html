63// Quic connection initialization goes like this
64// Initial -> Handshake -> 1-RTT
65// Initial - Establishes header protection key + Client\Server Hello + CIDs
66// Handshake - Establishes RTT keys + Handshake messages
67// 1-RTT - Application keys
68//[IP[UDP[QUICK_PKT[FRAME1[...], FRAME2[...], ...],[QUICK_PKT2.....]]]]
69//Loss detection is at the quic packet level, not per frame, uses counter
