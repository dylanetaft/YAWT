53// Initial -> Handshake -> 1-RTT
54// Initial - Establishes header protection key + Client\Server Hello + CIDs
55// Handshake - Establishes RTT keys + Handshake messages
56// 1-RTT - Application keys
57//[IP[UDP[QUICK_PKT[FRAME1[...], FRAME2[...], ...],[QUICK_PKT2.....]]]]
