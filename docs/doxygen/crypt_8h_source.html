14// QUIC encryption levels — index into level_keys[4] array.
15// These match gnutls_record_encryption_level_t values.
16//
17// Secret sources per level:
18// INITIAL: Derived from client's DCID + well-known salt (RFC 9001 §5.2).
19// No TLS handshake needed — both sides compute independently.
20// EARLY: 0-RTT secrets from TLS (not implemented).
21// HANDSHAKE: TLS handshake secrets, delivered by GnuTLS _on_secret callback.
22// APPLICATION: TLS application traffic secrets, delivered by GnuTLS _on_secret callback.
23//
24// For all levels, the same HKDF-Expand-Label step converts the traffic secret
25// into usable key/iv/hp:
26// key = HKDF-Expand-Label(secret, "quic key", "", key_len)
27// iv = HKDF-Expand-Label(secret, "quic iv", "", 12)
28// hp = HKDF-Expand-Label(secret, "quic hp", "", key_len)
29typedef enum {
30 YAWT_Q_LEVEL_INITIAL = 0,
31 YAWT_Q_LEVEL_EARLY = 1,
32 YAWT_Q_LEVEL_HANDSHAKE = 2,
33 YAWT_Q_LEVEL_APPLICATION = 3
34} YAWT_Q_Encryption_Level_t;

36// Per-level key state — tracks lifecycle of encryption keys at each level.

40typedef enum {
41 YAWT_Q_KEY_STATE_INACTIVE = 0,
