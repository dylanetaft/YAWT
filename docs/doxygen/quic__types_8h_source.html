50static inline void YAWT_q_cid_set(YAWT_Q_Cid_t *dst, const uint8_t *id, uint8_t len) {
51 if (len > 20) {
52 len = 20;
53 YAWT_LOG(YAWT_LOG_ERROR, "CID length %u exceeds max, truncating to 20", len);
54 }
55 dst->len = len;
56 memcpy(dst->id, id, len);
57}

65static inline const char *YAWT_q_cid_to_hex(const YAWT_Q_Cid_t *cid) {
66 static char buf[41];
67 memset(buf, 0, sizeof(buf));
68 for (uint8_t i = 0; i < cid->len && i < 20; i++) {
69 sprintf(buf + 2 * i, "%02x", cid->id[i]);
70 }
71 return buf;
72}
