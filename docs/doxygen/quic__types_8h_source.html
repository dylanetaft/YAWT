76static inline const char *YAWT_q_cid_to_hex(const YAWT_Q_Cid_t *cid) {
77 static char buf[41];
78 memset(buf, 0, sizeof(buf));
79 for (uint8_t i = 0; i < cid->len && i < 20; i++) {
80 sprintf(buf + 2 * i, "%02x", cid->id[i]);
81 }
82 return buf;
83}
