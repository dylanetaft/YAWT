56static inline void YAWT_q_cid_set(YAWT_Q_Cid_t *dst, const uint8_t *id, uint8_t len) {
57 if (len > 20) {
58 len = 20;
59 YAWT_LOG(YAWT_LOG_ERROR, "CID length %u exceeds max, truncating to 20", len);
60 }
61 dst->len = len;
62 memcpy(dst->id, id, len);
63}

71static inline const char *YAWT_q_cid_to_hex(const YAWT_Q_Cid_t *cid) {
72 static char buf[41];
73 memset(buf, 0, sizeof(buf));
74 for (uint8_t i = 0; i < cid->len && i < 20; i++) {
75 sprintf(buf + 2 * i, "%02x", cid->id[i]);
76 }
77 return buf;
78}
