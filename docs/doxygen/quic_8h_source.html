71// Quic connection initialization goes like this
72// Initial -> Handshake -> 1-RTT
73// Initial - Establishes header protection key + Client\Server Hello + CIDs
74// Handshake - Establishes RTT keys + Handshake messages
75// 1-RTT - Application keys
76//[IP[UDP[QUICK_PKT[FRAME1[...], FRAME2[...], ...],[QUICK_PKT2.....]]]]
77//Loss detection is at the quic packet level, not per frame, uses counter

127typedef struct {
128 uint64_t tx_count_bytes; // Pre-validation: cumulative wire bytes for anti-amplification (RFC 9000 §8.1). Post-validation: cumulative stream body bytes for connection flow control (RFC 9000 §4.1). Zeroed on address validation.
129 uint64_t rx_count_bytes; // Pre-validation: cumulative payload bytes for anti-amplification (RFC 9000 §8.1). Post-validation: cumulative stream body bytes for connection flow control (RFC 9000 §4.1). Zeroed on address validation.
130 // RFC 9000 Section 12.3 - Counters for each space, indexed by YAWT_Q_Encryption_Level_t
131 uint64_t next_pkt_num_tx[4]; // per encryption level: next TX packet number to send
132 uint64_t next_pkt_num_rx[4]; // per encryption level: high-water RX PN + 1 (drives PN reconstruction, RFC 9000 App. A); monotonic
133 // RFC 9000 §12.3 anti-replay / reorder window per space: a 128-PN bitmap across two words,
134 // anchored at rx_pn_win_base[level] (the PN of bit0 of rx_pn_win_a). A reordered PN within
135 // the window is accepted once; duplicates and PNs below the window are discarded.
136 uint64_t rx_pn_win_a[4]; // bitmap for PNs [base .. base+63]
137 uint64_t rx_pn_win_b[4]; // bitmap for PNs [base+64 .. base+127]
138 uint64_t rx_pn_win_base[4]; // PN represented by bit0 of rx_pn_win_a[level]
139 uint64_t cid_seq_num; // highest NEW_CONNECTION_ID seq_num seen
140 double last_rx; // ev_now() timestamp of last packet received
141 double last_tx; // ev_now() timestamp of last packet sent
142 double closing_at; // 0 = not closing, else timestamp when close period started
143} YAWT_Q_ConnectionStats_t;
