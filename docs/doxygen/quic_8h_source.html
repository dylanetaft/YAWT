40// Quic connection initialization goes like this
41// Initial -> Handshake -> 1-RTT
42// Initial - Establishes header protection key + Client\Server Hello + CIDs
43// Handshake - Establishes RTT keys + Handshake messages
44// 1-RTT - Application keys
45//[IP[UDP[QUICK_PKT[FRAME1[...], FRAME2[...], ...],[QUICK_PKT2.....]]]]
46//Loss detection is at the quic packet level, not per frame, uses counter
