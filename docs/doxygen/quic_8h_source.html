48// Quic connection initialization goes like this
49// Initial -> Handshake -> 1-RTT
50// Initial - Establishes header protection key + Client\Server Hello + CIDs
51// Handshake - Establishes RTT keys + Handshake messages
52// 1-RTT - Application keys
53//[IP[UDP[QUICK_PKT[FRAME1[...], FRAME2[...], ...],[QUICK_PKT2.....]]]]
54//Loss detection is at the quic packet level, not per frame, uses counter
